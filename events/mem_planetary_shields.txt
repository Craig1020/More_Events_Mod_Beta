	id = mem_planetary_shields.3
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		# check for any_shield building
		IF = {
			limit = {
				any_tile = {
					OR = {
						has_building = building_autochthon_monument
						has_building = building_heritage_site
					}
					is_ruined = no
				}
			}
			random_tile = {
				limit = {
					OR = {
						has_building = building_autochthon_monument
						has_building = building_heritage_site
					}
					is_ruined = no
				}
				set_ruined = yes
			}
		}
	}
}
